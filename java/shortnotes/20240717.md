# 代码收集
#### 1. Stream API代码

```java
import java.util.*;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class StreamDemo {
    //判空-Nullable，使用ifPresent方法对集合进行判空操作
    public void ifPresent() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        //判空
        Optional.ofNullable(list).ifPresent(arr -> arr.forEach(s -> {
            System.out.println(s);
        }));
    }

    //过滤-Filtering，使用filter方法对集合进行筛选
    public void filtering() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        List<String> listFiltered = list.stream()
                .filter(s -> s.startsWith("A"))
                .toList();
    }

    //映射-Mapping，使用map将每个元素转换成另一种形式
    public void mapping() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        Set<String> setMapped = list.stream()
                .map(s -> s + "1")
                .collect(Collectors.toSet());
    }

    //排序-Sorting，通过sorted方法对集合进行排序
    public void sorting() {
        List<String> list = Arrays.asList("Charlie", "Alic", "David", "Bob");
        Stream<String> listSorted = list.stream()
                .sorted(String::compareTo);

    }

    //查找-Finding，通过findFirst或者findAny方法查找元素
    public void finding() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        Optional<String>
                found = list.stream().findFirst();
    }

    //匹配-Matching，通过anyMatch，allMatch或者noneMatch方法查看集合中元素是否符合某个条件
    public void matching() {
        boolean anyMatch = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .stream().anyMatch(s -> s.indexOf("D") == 0);

        boolean allMatch = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .stream().allMatch(s -> s.length() > 3);

        boolean noneMatch = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .stream().noneMatch(s -> s.length() > 5);
    }

    //计数-Counting,通过count方法，计数数组中的元素
    public void counting() {
        long count = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .stream().count();
    }

    //归约-Reducing，通过reduce方法将集合中的元素进行计算
    public void reducing() {
        int sum = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .stream().mapToInt(s -> s.length())
                .sum();
    }

    //收集-Collecting，通过collect方法将集合类型进行转换
    public void collecting() {
        Set<String> set = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .stream().collect(Collectors.toSet());

    }

    //并行流-Parallel Streams，通过parallelStream方法使用并行流来利用多核处理器加速集合操作
    public void parallelStreams() {
        Set<String> set = Arrays.asList("Alic", "Bob", "Charlie", "David")
                .parallelStream().filter(s -> s.length() > 3)
                .collect(Collectors.toSet());
    }


}
```
