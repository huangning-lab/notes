#### 1. Stream API代码

```java
import java.util.*;
import java.util.stream.Collectors;
import java.util.stream.Stream;

public class StreamDemo {
    //遍历非空数组
    public void ifPresent() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        //判空
        Optional.ofNullable(list).ifPresent(arr -> arr.forEach(s -> {
            System.out.println(s);
        }));
    }

    //过滤-Filtering
    public void filtering() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        List<String> listFiltered = list.stream()
                .filter(s -> s.startsWith("A"))
                .toList();
    }

    //映射-Mapping
    public void mapping() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        Set<String> setMapped = list.stream()
                .map(s -> s + "1")
                .collect(Collectors.toSet());
    }

    //排序-Sorting
    public void sorting() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
        Stream<String> listSorted = list.stream()
                .sorted(String::compareTo);

    }

    //查找-Finding
    public void finding() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
    }

    //匹配-Matching
    public void matching() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
    }

    //计数-Counting
    public void counting() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
    }

    //归约-Reducing
    public void reducing() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
    }

    //收集-Collecting
    public void collecting() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
    }

    //并行流-Parallel Streams
    public void parallelStreams() {
        List<String> list = Arrays.asList("Alic", "Bob", "Charlie", "David");
    }


}
```
